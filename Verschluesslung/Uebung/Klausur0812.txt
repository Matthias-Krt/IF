String decrypt(String klarTxt, int key) {
    String chiffre = "";
    char klarCh;
    int charNum;
    int len = klarTxt.length();

    for(int i = 0; i < len; i++) {
        klarCh = klarTxt.charAt(i);

        charNum = (int) klarCh - 65;
        charNum += key;
        charNum %= 26;

        chiffre += (char) (charNum + 65);
    }

    return chiffre;
}
